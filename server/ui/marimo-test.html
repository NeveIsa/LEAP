import marimo as mo
import numpy as np
import matplotlib.pyplot as plt
# Create some sample data
x = np.linspace(0, 10, 100)
y = np.sin(x)
# Plot the data
plt.figure(figsize=(10, 6))
plt.plot(x, y, 'b-', linewidth=2)
plt.title('Simple Sine Wave', color='white')
plt.xlabel('x', color='white')
plt.ylabel('sin(x)', color='white')
plt.grid(True, alpha=0.3)
plt.show()

import marimo as mo
# Create an interactive slider
slider = mo.ui.slider(1, 10, value=5)
slider

f"Slider value: {slider.value}"
# Create a simple visualization
import matplotlib.pyplot as plt
import numpy as np

x = np.linspace(0, 10, 100)
y = np.sin(x * slider.value)

plt.figure(figsize=(10, 6))
plt.plot(x, y, 'g-', linewidth=2)
plt.title(f'Sine Wave with Frequency {slider.value}', color='white')
plt.xlabel('x', color='white')
plt.ylabel('sin(x)', color='white')
plt.grid(True, alpha=0.3)
plt.show()
